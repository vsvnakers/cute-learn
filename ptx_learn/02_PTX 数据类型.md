# 第二课：PTX 数据类型

## 2.1 PTX 数据类型概览

### 基本数据类型

```
┌─────────────────────────────────────────────────────────────────┐
│  PTX 数据类型分类                                                │
├─────────────────────────────────────────────────────────────────┤
│                                                                 │
│  1. 整数类型                                                    │
│     .u8    - 8 位无符号整数                                      │
│     .u16   - 16 位无符号整数                                     │
│     .u32   - 32 位无符号整数                                     │
│     .u64   - 64 位无符号整数                                     │
│     .s8    - 8 位有符号整数                                      │
│     .s16   - 16 位有符号整数                                     │
│     .s32   - 32 位有符号整数                                     │
│     .s64   - 64 位有符号整数                                     │
│                                                                 │
│  2. 浮点类型                                                    │
│     .f16   - 16 位半精度浮点数 (FP16)                           │
│     .bf16  - 16 位 Brain 浮点数 (BF16)                          │
│     .f32   - 32 位单精度浮点数 (FP32)                           │
│     .f64   - 64 位双精度浮点数 (FP64)                           │
│                                                                 │
│  3. 位类型                                                      │
│     .b1    - 1 位（谓词）                                        │
│     .b8    - 8 位                                                │

│     .b64   - 64 位                                               │
│                                                                 │
│  4. 向量类型                                                    │
│     .v2.u32  - 2×u32 向量 (64 位)                               │

│                                                                 │
└─────────────────────────────────────────────────────────────────┘
```

## 2.2 整数类型详解

### 整数类型表示

```ptx
// 声明整数寄存器
.reg .u8  %ru8;      // 8 位无符号
.reg .u16 %ru16;     // 16 位无符号
.reg .u32 %ru32;     // 32 位无符号
.reg .u64 %ru64;     // 64 位无符号

.reg .s8  %rs8;      // 8 位有符号
.reg .s16 %rs16;     // 16 位有符号
.reg .s32 %rs32;     // 32 位有符号
.reg .s64 %rs64;     // 64 位有符号

// 整数运算示例
add.u32 %ru32_1, %ru32_2, %ru32_3;    // 无符号加法
add.s32 %rs32_1, %rs32_2, %rs32_3;    // 有符号加法
mul.lo.u32 %ru32_1, %ru32_2, %ru32_3; // 乘法（取低位）
mul.hi.u32 %ru32_1, %ru32_2, %ru32_3; // 乘法（取高位）
```

### 整数溢出处理

```ptx
// 32 位乘法溢出示例
// 假设 %r1 = 0xFFFFFFFF, %r2 = 2

mul.lo.u32 %r3, %r1, %r2;   // %r3 = 0xFFFFFFFE (低位)
mul.hi.u32 %r4, %r1, %r2;   // %r4 = 0x00000001 (高位)

// 完整 64 位结果：%r4:%r3 = 0x00000001FFFFFFFE

// 带饱和的加法（避免溢出）
add.sat.u32 %r5, %r1, %r2;  // 溢出时饱和到最大值
```

### 整数比较

```ptx
// 无符号比较
setp.lt.u32 %p1, %r1, %r2;    // %p1 = (%r1 < %r2)
setp.ge.u32 %p2, %r1, %r2;    // %p2 = (%r1 >= %r2)
setp.eq.u32 %p3, %r1, %r2;    // %p3 = (%r1 == %r2)

// 有符号比较
setp.lt.s32 %p4, %r1, %r2;    // %p4 = (%r1 < %r2) 有符号
setp.gt.s32 %p5, %r1, %r2;    // %p5 = (%r1 > %r2) 有符号

// 比较并设置寄存器
set.lt.u32 %r3, %r1, %r2;     // %r3 = (%r1 < %r2) ? 1 : 0
```

## 2.3 浮点类型详解

### 浮点类型表示

```ptx
// 声明浮点寄存器
.reg .f16 %rf16;       // 半精度
.reg .bf16 %rbf16;     // Brain 浮点
.reg .f32 %rf32;       // 单精度
.reg .f64 %rf64;       // 双精度

// 浮点运算示例
add.f32 %f1, %f2, %f3;           // 单精度加法
mul.f32 %f1, %f2, %f3;           // 单精度乘法
fma.rn.f32 %f1, %f2, %f3, %f4;   // 乘加（舍入到最近）
div.f32 %f1, %f2, %f3;           // 单精度除法
sqrt.f32 %f1, %f2;               // 平方根
```

### 浮点舍入模式

```ptx
// 四种 IEEE 舍入模式

// .rn - Round to Nearest (默认)
fma.rn.f32 %f1, %f2, %f3, %f4;   // 舍入到最近

// .rz - Round toward Zero
fma.rz.f32 %f1, %f2, %f3, %f4;   // 向零舍入

// .rm - Round toward -Infinity
fma.rm.f32 %f1, %f2, %f3, %f4;   // 向负无穷舍入

// .rp - Round toward +Infinity
fma.rp.f32 %f1, %f2, %f3, %f4;   // 向正无穷舍入
```

### 浮点特殊值处理

```ptx
// FTZ (Flush To Zero) - 非规格化数视为零
add.ftz.f32 %f1, %f2, %f3;    // 输入/输出 FTZ

// 近似计算（更快但精度略低）
sqrt.approx.f32 %f1, %f2;     // 近似平方根
rcp.approx.f32 %f1, %f2;      // 近似倒数
sin.approx.f32 %f1, %f2;      // 近似正弦

// 精确计算
sqrt.rn.f32 %f1, %f2;         // 精确平方根
rcp.approx.ftz.f32 %f1, %f2;  // 近似倒数 + FTZ
```

### 浮点比较

```ptx
// 浮点比较（注意 NaN 处理）
setp.lt.f32 %p1, %f1, %f2;    // %f1 < %f2
setp.eq.f32 %p2, %f1, %f2;    // %f1 == %f2
setp.le.f32 %p3, %f1, %f2;    // %f1 <= %f2

// 有序比较（排除 NaN）
setp.ltu.f32 %p4, %f1, %f2;   // %f1 < %f2 (unordered safe)

// 特殊值检查
setp.nan.f32 %p5, %f1;        // 检查 NaN
setp.inf.f32 %p6, %f1;        // 检查无穷大
```

## 2.4 位类型详解

### 位类型用于位操作

```ptx
// 声明位寄存器
.reg .b8  %b8_reg;
.reg .b16 %b16_reg;
.reg .b32 %b32_reg;
.reg .b64 %b64_reg;

// 位运算
and.b32 %b32_1, %b32_2, %b32_3;   // 按位与
or.b32  %b32_1, %b32_2, %b32_3;   // 按位或
xor.b32 %b32_1, %b32_2, %b32_3;   // 按位异或
not.b32 %b32_1, %b32_2;           // 按位取反

// 位移
shl.b32 %b32_1, %b32_2, %r1;      // 左移
shr.b32 %b32_1, %b32_2, %r1;      // 逻辑右移
ashr.b32 %b32_1, %b32_2, %r1;     // 算术右移

// 位查找
bfind.u32 %r1, %r2;               // 找第一个为 1 的位
ffs.u32 %r1, %r2;                 // 找第一个为 1 的位（从 1 开始）
clz.u32 %r1, %r2;                 // 前导零计数
popc.u32 %r1, %r2;                // 1 的个数计数
```

### 位类型与整数类型互换

```ptx
// 位类型和整数类型可以互换使用
.reg .b32 %b;
.reg .u32 %u;

mov.b32 %b, %u;      // u32 → b32
mov.b32 %u, %b;      // b32 → u32

// 位操作后转回整数
and.b32 %b, %u, 0xFF;    // 取低 8 位
cvt.u32.u8 %u, %b;       // b32 → u32（只取低 8 位）
```

## 2.5 向量类型详解

### 向量类型声明

```ptx
// 声明向量寄存器
.reg .v2.u16 %v2u16;     // 2×u16 向量
.reg .v2.u32 %v2u32;     // 2×u32 向量
.reg .v4.u8  %v4u8;      // 4×u8 向量
.reg .v4.u32 %v4u32;     // 4×u32 向量
.reg .v4.f32 %v4f32;     // 4×f32 向量

// 向量加载
ld.global.v4.u8 {%v4u8}, [%rd1];    // 一次加载 4 个 u8
ld.global.v2.u32 {%v2u32}, [%rd1];  // 一次加载 2 个 u32

// 向量存储
st.global.v4.f32 [%rd1], {%v4f32};  // 一次存储 4 个 f32
```

### 向量元素提取

```ptx
// 从向量提取元素
.reg .v4.u8 %v;
.reg .u8 %e0, %e1, %e2, %e3;

// 方法 1: 使用 mov
mov.u8 %e0, %v.x;    // 提取第 0 个元素
mov.u8 %e1, %v.y;    // 提取第 1 个元素
mov.u8 %e2, %v.z;    // 提取第 2 个元素
mov.u8 %e3, %v.w;    // 提取第 3 个元素

// 方法 2: 使用合并/拆分
mergelo.b32 %r1, %v, %v;   // 合并低 16 位
mergehi.b32 %r2, %v, %v;   // 合并高 16 位
```

### 向量操作示例

```ptx
// 4×u8 向量加法（并行处理 4 个字节）
.reg .v4.u8 %v1, %v2, %vr;

// 加载 4 个字节
ld.global.v4.u8 {%v1}, [%rd1];
ld.global.v4.u8 {%v2}, [%rd2];

// 向量加法（需要使用 add 指令的向量版本）
// 注意：PTX 不直接支持向量算术，需要拆分后计算

// 更常见的用法是用于内存传输
ld.global.v4.u32 {%vr}, [%rd1];  // 128 位加载
st.global.v4.u32 [%rd2], {%vr};  // 128 位存储
```

## 2.6 类型转换

### 整数类型转换

```ptx
// 扩展转换
cvt.u16.u8  %r1, %r2;     // u8 → u16 (零扩展)
cvt.u32.u16 %r1, %r2;     // u16 → u32 (零扩展)
cvt.u64.u32 %rd1, %r2;    // u32 → u64 (零扩展)

cvt.s16.s8  %r1, %r2;     // s8 → s16 (符号扩展)
cvt.s32.s8  %r1, %r2;     // s8 → s32 (符号扩展)

// 截断转换
cvt.u8.u16  %r1, %r2;     // u16 → u8 (截断)
cvt.u16.u32 %r1, %r2;     // u32 → u16 (截断)
cvt.u32.u64 %r1, %rd2;    // u64 → u32 (截断)
```

### 浮点类型转换

```ptx
// 浮点→整数
cvt.u32.f32 %r1, %f1;     // f32 → u32
cvt.s32.f32 %r1, %f1;     // f32 → s32
cvt.rzi.u32.f32 %r1, %f1; // f32 → u32 (向零舍入)

// 整数→浮点
cvt.f32.u32 %f1, %r1;     // u32 → f32
cvt.f32.s32 %f1, %r1;     // s32 → f32

// 浮点→浮点
cvt.f16.f32 %h1, %f1;     // f32 → f16
cvt.f32.f16 %f1, %h1;     // f16 → f32
cvt.f64.f32 %fd1, %f1;    // f32 → f64
cvt.f32.f64 %f1, %fd1;    // f64 → f32
```

### 浮点舍入转换

```ptx
// 带舍入模式的浮点转换
cvt.rn.f16.f32 %h1, %f1;   // f32→f16 (最近)
cvt.rz.f16.f32 %h1, %f1;   // f32→f16 (向零)
cvt.rm.f16.f32 %h1, %f1;   // f32→f16 (负无穷)
cvt.rp.f16.f32 %h1, %f1;   // f32→f16 (正无穷)

// 饱和转换（限制在目标类型范围内）
cvt.sat.u32.f32 %r1, %f1;  // f32→u32 (饱和)
```

### 位模式转换

```ptx
// 位模式 reinterpret（不改变位，只改变解释方式）
mov.b32 %f1, %r1;     // u32 → f32 (位模式不变)
mov.b32 %r1, %f1;     // f32 → u32 (位模式不变)

mov.b64 %fd1, %rd1;   // u64 → f64
mov.b64 %rd1, %fd1;   // f64 → u64

// 示例：float 到 int 的位模式转换
.reg .f32 %f;
.reg .u32 %u;
mov.b32 %u, %f;       // 获取 float 的位表示
```

## 2.7 指针类型

### 指针声明和操作

```ptx
// 指针使用 .u64 类型
.reg .u64 %ptr;

// 加载参数指针
ld.param.u64 %ptr, [input_ptr];

// 指针算术
add.u64 %ptr2, %ptr, 4;     // 指针 + 4 字节
mul.u64 %offset, %idx, 4;   // 索引 × 4
add.u64 %ptr3, %ptr, %offset; // 基地址 + 偏移

// 通过指针加载/存储
ld.global.f32 %f1, [%ptr];      // 从指针加载 f32
st.global.f32 [%ptr], %f1;      // 存储 f32 到指针
```

### 地址空间转换

```ptx
// 不同地址空间的指针转换
cvta.global.u64 %rd1, %rd2;     // 转换为全局地址
cvta.shared.u64 %rd1, %rd2;     // 转换为共享地址
cvta.const.u64 %rd1, %rd2;      // 转换为常量地址
cvta.local.u64 %rd1, %rd2;      // 转换为本地地址

// 示例：共享内存指针
shared .u32 shared_data[256];
.entry kernel() {
    .reg .u64 %ptr;
    cvta.shared.u64 %ptr, shared_data;
    st.shared.u32 [%ptr], 42;
}
```

## 2.8 类型大小和对齐

### 类型大小

```
类型        大小（字节）
─────────────────────────
.u8/.s8     1
.u16/.s16   2
.u32/.s32   4
.u64/.s64   8
.f16        2
.bf16       2
.f32        4
.f64        8
.v2.u32     8
.v4.u32     16
.v4.f32     16
```

### 对齐要求

```ptx
// 显式对齐声明
.global .align 4 .u32 aligned_var;      // 4 字节对齐
.global .align 8 .u64 aligned_var2;     // 8 字节对齐
.global .align 16 .v4.u32 vec_var;      // 16 字节对齐

// 内存访问对齐
ld.global.aligned.f32 %f1, [%rd1];  // 对齐加载（更快）
ld.global.f32 %f1, [%rd1];          // 非对齐加载
```

## 2.9 小结

| 数据类型 | 代表符号 | 典型用途 |
|---------|---------|---------|
| .u32/.s32 | %r | 索引、计数 |
| .u64 | %rd | 指针、地址 |
| .f32 | %f | 浮点计算 |
| .f64 | %fd | 双精度计算 |
| .b32 | %b | 位操作 |
| .v4.u8 | %v | 向量加载/存储 |
| .pred | %p | 条件判断 |
